async function updateUI() { const res = await fetch('data.json'); const kb = await res.json(); // Update the Metric Card document.getElementById('recent-count').innerText = kb.count; document.getElementById('sub-text').innerText = `New findings ${kb.time_window}`; // Update the Table const table = document.getElementById('vuln-table'); table.innerHTML = kb.data.map(item => ` ${item.qid} ${item.title} ${item.severity} ${item.published} `).join(''); }
